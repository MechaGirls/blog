function withFetchImage(InnerComponent) {
  return {
    data() {
      return {
        url: 'http://via.placeholder.com/200x200'
      };
    },
    created() {
      setTimeout(() => {
        this.url = '.../avatar.jpg'
      }, 1000);
    },
    render(h) {
      const attributes = {
        attrs: this.$attrs,
        listeners: this.$listeners,
        style: this.$style,
        class: this.$class
      };
      return (
        <InnerComponent src={this.url} {...attributes} />
      );
    }
  }
};
